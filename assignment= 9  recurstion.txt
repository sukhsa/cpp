//Q.1 Write a recursive function to print numbers from 1 to 15.
#include <bits/stdc++.h>
using namespace std;
void recursion(int n)
{
    if(n<=0)
        return;
   
    recursion(n-1);
    cout<<n<<" ";
}
int main()
{
    int n;
    cout<<"Enter the value of n:\n";
    cin>>n;
    recursion(n);
}
